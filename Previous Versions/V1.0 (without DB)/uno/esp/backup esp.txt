#include <WiFi.h>
#include <PubSubClient.h>
#include <WebServer.h>

// ---------- WiFi ----------
const char* ssid = "ACCESS DENIED";
const char* password = "Power@2026SLT";

// ---------- MQTT ----------
const char* mqtt_server = "broker.hivemq.com";
WiFiClient espClient;
PubSubClient client(espClient);

// ---------- Web Server ----------
WebServer server(80);

// ---------- Pins ----------
#define BUZZER     2
#define LED_POWER  19
#define LED_WIFI   21

// ---------- Lift States ----------
bool liftLatched[3][2] = {
  {false,false},
  {false,false},
  {false,false}
};

// ---------- UNO Health ----------
unsigned long lastSeenUNO[3] = {0,0,0};
const unsigned long UNO_TIMEOUT_MS = 15000;

// ---------- Alarm latch ----------
bool alarmLatched = false;

// ---------- Buzzer timing ----------
const unsigned long BUZZER_ON_TIME  = 1000;
const unsigned long BUZZER_OFF_TIME = 2000;
unsigned long lastBuzzerChange = 0;
bool buzzerState = false;

// ---------- WiFi LED ----------
const unsigned long WIFI_BLINK_INTERVAL = 500;
bool wifiLedState = false;
unsigned long lastWifiBlink = 0;

// ---------- UNO status helpers ----------
String unoStatusText(int unoIndex){
  if (lastSeenUNO[unoIndex] == 0) return "Deactive";
  unsigned long age = millis() - lastSeenUNO[unoIndex];
  return (age <= UNO_TIMEOUT_MS) ? "Active" : "Deactive";
}

String unoStatusColor(int unoIndex){
  if (lastSeenUNO[unoIndex] == 0) return "red";
  unsigned long age = millis() - lastSeenUNO[unoIndex];
  return (age <= UNO_TIMEOUT_MS) ? "green" : "red";
}

// ---------- Update buzzer ----------
void updateBuzzer(){
  if (!alarmLatched) {
    digitalWrite(BUZZER, LOW);
    buzzerState = false;
    lastBuzzerChange = millis();
    return;
  }

  unsigned long now = millis();
  if (buzzerState && now - lastBuzzerChange >= BUZZER_ON_TIME) {
    buzzerState = false;
    lastBuzzerChange = now;
    digitalWrite(BUZZER, LOW);
  }
  else if (!buzzerState && now - lastBuzzerChange >= BUZZER_OFF_TIME) {
    buzzerState = true;
    lastBuzzerChange = now;
    digitalWrite(BUZZER, HIGH);
  }
}

// ---------- WiFi LED ----------
void updateWifiLed(){
  if (WiFi.status() == WL_CONNECTED) {
    digitalWrite(LED_WIFI, HIGH);
    return;
  }

  if (millis() - lastWifiBlink >= WIFI_BLINK_INTERVAL) {
    lastWifiBlink = millis();
    wifiLedState = !wifiLedState;
    digitalWrite(LED_WIFI, wifiLedState);
  }
}

// ---------- MQTT Callback ----------
void mqttCallback(char* topic, byte* payload, unsigned int length) {
  String msg="";
  for(unsigned int i=0;i<length;i++) msg+=(char)payload[i];
  msg.trim();

  String t = String(topic);

  // Update UNO "last seen" when ANY message arrives from that UNO
  if (t.startsWith("test/relay/uno1/")) lastSeenUNO[0] = millis();
  else if (t.startsWith("test/relay/uno2/")) lastSeenUNO[1] = millis();
  else if (t.startsWith("test/relay/uno3/")) lastSeenUNO[2] = millis();

  // keep original behavior: latch only when ON
  if(msg!="ON") return;

  if      (t=="test/relay/uno1/lift1") liftLatched[0][0]=true;
  else if (t=="test/relay/uno1/lift2") liftLatched[0][1]=true;
  else if (t=="test/relay/uno2/lift3") liftLatched[1][0]=true;
  else if (t=="test/relay/uno2/lift4") liftLatched[1][1]=true;
  else if (t=="test/relay/uno3/lift5") liftLatched[2][0]=true;
  else if (t=="test/relay/uno3/lift6") liftLatched[2][1]=true;

  alarmLatched = true;
}

// ---------- MQTT reconnect ----------
void reconnectMQTT(){
  while(!client.connected()){
    if(client.connect("ESP32_Dashboard")){
      client.subscribe("test/relay/uno1/lift1");
      client.subscribe("test/relay/uno1/lift2");
      client.subscribe("test/relay/uno2/lift3");
      client.subscribe("test/relay/uno2/lift4");
      client.subscribe("test/relay/uno3/lift5");
      client.subscribe("test/relay/uno3/lift6");
      client.subscribe("test/relay/uno1/status");
      client.subscribe("test/relay/uno2/status");
      client.subscribe("test/relay/uno3/status");
    } else {
      delay(2000);
    }
  }
}

// ---------- Reset ----------
void handleReset(){
  int g = server.hasArg("g") ? server.arg("g").toInt() : -1;
  int l = server.hasArg("l") ? server.arg("l").toInt() : -1;

  if (g < 0 || g > 2 || l < 0 || l > 1) {
    server.send(400, "text/plain", "Bad args");
    return;
  }

  liftLatched[g][l] = false;

  // recompute alarmLatched
  alarmLatched = false;
  for(int i=0;i<3;i++){
    for(int j=0;j<2;j++){
      if(liftLatched[i][j]) alarmLatched = true;
    }
  }

  server.send(200, "text/plain", "OK");
}

// ---------- JSON state endpoint (NO full page refresh) ----------
void handleState() {
  String json = "{";

  // Lift latched state
  json += "\"lift\":[";
  for (int i = 0; i < 3; i++) {
    json += "[";
    for (int j = 0; j < 2; j++) {
      json += (liftLatched[i][j] ? "1" : "0");
      if (j < 1) json += ",";
    }
    json += "]";
    if (i < 2) json += ",";
  }
  json += "],";

  // UNO health state
  json += "\"uno\":[";
  for (int i = 0; i < 3; i++) {
    bool active = false;
    if (lastSeenUNO[i] != 0 && (millis() - lastSeenUNO[i] <= UNO_TIMEOUT_MS)) active = true;
    json += (active ? "1" : "0");
    if (i < 2) json += ",";
  }
  json += "]";

  json += "}";

  server.send(200, "application/json", json);
}

// ---------- Build lift ----------
String buildLift(int num,int g,int l){
  String color = liftLatched[g][l] ? "red" : "green";

  String s="";
  s += "<div class='lift'>";
  s += "<div id='lift_" + String(g) + "_" + String(l) + "' class='circle' style='background:" + color + "'></div>";
  s += "<div class='label'>Lift " + String(num) + "</div>";
  s += "<button class='btn' onclick='resetLift(" + String(g) + "," + String(l) + ")'>RESET</button>";
  s += "</div>";
  return s;
}

}

// ---------- Setup ----------
void setup(){
  Serial.begin(115200);
  delay(1500);

  pinMode(BUZZER,OUTPUT);
  pinMode(LED_POWER,OUTPUT);
  pinMode(LED_WIFI,OUTPUT);

  digitalWrite(LED_POWER,HIGH);
  digitalWrite(BUZZER,LOW);

  WiFi.begin(ssid,password);

  // ===== PRINT DASHBOARD LINK =====
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println();
  Serial.println("====================================");
  Serial.println("ESP32 Lift Dashboard Ready");
  Serial.print("Dashboard URL: http://");
  Serial.println(WiFi.localIP());
  Serial.println("====================================");

  client.setServer(mqtt_server,1883);
  client.setCallback(mqttCallback);

  server.on("/", handleWeb);
  server.on("/reset", handleReset);
  server.on("/state", handleState);   // NEW: state endpoint
  server.begin();
}

// ---------- Loop ----------
void loop(){
  updateWifiLed();

  if(WiFi.status()==WL_CONNECTED){
    if(!client.connected()) reconnectMQTT();
    client.loop();
  }

  server.handleClient();
  updateBuzzer();
}
